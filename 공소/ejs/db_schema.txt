create table groupInfo
(
    id              bigint(10)   auto_increment,
    groupid
)


select mdl_groups.name,groupid, firstname, lastname
from mdl_groups_members, mdl_user,mdl_groups
where mdl_groups_members.userid = mdl_user.id and mdl_groups_members.groupid=1 and mdl_groups_members.groupid = mdl_groups.id;



select name, firstname, lastname
from mdl_groups, mdl_user
where (select groupid, firstname, lastname
from mdl_groups_members, mdl_user
where mdl_groups_members.userid = mdl_user.id and mdl_groups_members.groupid=1).groupid = mdl_groups.id;
